# Assigning mortal trait at start and cycling

AB_Mortals_events.4 = {
	type = character_event
	hidden = yes

	trigger = {
		is_alive = yes
		NOR = {
			has_trait = vampire
			has_trait = werewolf
			has_trait = mage
			has_trait = demon
			has_trait = mortal	
		}
	}
	
	immediate = {
		hidden_effect = {
		add_trait = mortal
		}
	}
}
