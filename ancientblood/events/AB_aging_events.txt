namespace = AB_aging_events

# Vampire ladder

AB_aging_events.1 = {
	type = character_event
	title = "AB_aging_events.1.t"
	desc = "AB_aging_events.1.desc"
	picture = greek_court
	
	left_portrait = root
	
	trigger = {
		is_alive = yes
		age > 250
		has_trait = neonate
		has_trait = vampire
	}
	
	weight_multiplier = {
		modifier = {
			factor = 5
			age > 270
		}
		modifier = {
			factor = 10
			age > 295
		}
	}
		
	option = {
		name = "AB_aging_events.1.a"
		
		remove_trait = neonate
		add_trait = ancilla
	}
}
